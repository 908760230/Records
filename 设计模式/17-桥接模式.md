# 桥接模式





## 定义



将抽象部分与它的实现部分分离，使它们都可以独立地变化

![](F:\GithubOpenSource\Records\设计模式\image\桥接模式.png)





## 代码

```c++
#include<iostream>

using namespace std;


class Implementor
{
public:
	virtual void Operation() =0;
};

class ConcreteImplementorA : public Implementor
{
public:
	void Operation() override {
		cout << "具体实现A的方法执行" << endl;
	}
};

class ConcreteImplementorB : public Implementor
{
public:
	void Operation() override {
		cout << "具体实现B的方法执行" << endl;
	}
};


class Abstraction
{
public:
	virtual ~Abstraction() {
		if (imp) delete imp;
	}
	void setImplementor(Implementor *tmp) {
		if (imp) delete imp;
		imp = tmp;
	}
	virtual void Operation() {
		imp->Operation();
	}
protected:
	Implementor *imp;
};

class RefinedAbstraction : public Abstraction
{
public:
	void Operation() override {
		imp->Operation();
	}
};

int main() {

	Abstraction *abs = new RefinedAbstraction();
	abs->setImplementor(new ConcreteImplementorA());
	abs->Operation();

	abs->setImplementor(new ConcreteImplementorB());
	abs->Operation();
	return 0;
}
```
